41/*
42 * @brief Distribui as cartas e inicializa as mãos dos jogadores
43 */
44void Burro::InicializaJogadores()
45{
46 switch(quantidadeJogadores)
47 {
48 case 4:
49 jogadores.push_back(Jogador(4));
50
51 case 3:
52 jogadores.push_back(Jogador(3));
53
54 case 2:
55 jogadores.push_back(Jogador(2));
56 jogadores.push_back(Jogador(1));
57 break;
58
59 default:
60 ThrowJogadoresException();
61 break;
62 }
63
64 reverse(jogadores.begin(), jogadores.end());
65
66 // Iniciar mao dos jogadores
67 for (auto j = jogadores.begin(); j != jogadores.end(); ++j)
68 {
69 j->IniciarMao(Baralho(cartas.Distribuir(5)));
70
71#ifdef DEBUG_Burro_InicializaJogadores
72 printf("\nJogador: %d", j->GetNumero());
73 printf("\nMao Inicial: ");
74 j->ImprimirMao();
75 Wait();
76#endif
77
78 }
79}

195/*
196 * @brief Loop da partida, responsável por processar e controlar o fluxo do jogo
197 */
198void Burro::LoopPartida()
199{
200#ifdef DEBUG_Burro_LoopPartida
201 cartas.ImprimeBaralho();
202 Wait();
203#endif
204
205 int posicaoCarta = 0;
206 int jogadorMaiorCarta = 0;
207 Carta maiorCarta;
208
209 // Loop do jogo
210 while(true)
211 {
212 ImprimirMesa();
213 printf("\nVez do Jogador %d", jogadorAtual + 1);
214 jogadores[jogadorAtual].ImprimirMao();
215 int cartasCompradas = 0;
216 while(!jogadores[jogadorAtual].CartaDisponivel(mesa.GetNaipe()))
217 {
218 if(cartas.QuantidadeCartas() == 0)
219 {
220 FimDeJogo(JogadorComMenosCartas());
221 return;
222 }
223
224 if(cartasCompradas == 0)
225 {
226 printf("\nVoce nao tem cartas desse naipe: hora de comprar");
227 Wait();
228 }
229
230 Carta cartaComprada = ComprarCartaDoBaralho();
231 jogadores[jogadorAtual].AdicionarCartaNaMao(cartaComprada);
232
233#ifdef DEBUG_Burro_LoopPartida
234 printf("\nComprando carta: ");
235 cartaComprada.ImprimeCarta();
236#endif
237 ++cartasCompradas;
238 }
239
240 if(cartasCompradas != 0)
241 {
242 printf("\nVoce comprou %d cartas", cartasCompradas);
243 jogadores[jogadorAtual].ImprimirMao();
244 }
245
246 posicaoCarta = jogadores[jogadorAtual].EscolherCartaParaJogar(mesa.GetNaipe());
247 Carta c = jogadores[jogadorAtual].JogarCarta(posicaoCarta);
248 mesa.AdicionarCarta(c, jogadorAtual);
249
250#ifdef DEBUG_Burro_LoopPartida
251 printf("\nCarta escolhida pelo jogador: ");
252 c.ImprimeCarta();
253 printf("\nQuantidade de cartas: %d", jogadores[jogadorAtual].QuantidadeCartasMao());
254 printf("\nMao do jogador apos a rodada: ");
255 jogadores[jogadorAtual].ImprimirMao();
256 Wait();
257#endif
258
259 if (ChecaFimJogo())
260 {
261 FimDeJogo(jogadorAtual + 1);
262 return;
263 }
264
265 if(mesa.QuantidadeCartas() == 1)
266 {
267 jogadorMaiorCarta = jogadorAtual;
268 maiorCarta = c;
269 }
270 else
271 {
272 if(c > maiorCarta)
273 {
274 maiorCarta = c;
275 jogadorMaiorCarta = jogadorAtual;
276 }
277 }
278
279 if(ChecaFimRodada())
280 {
281 FinalizarRodada(jogadorMaiorCarta);
282 mesa.Limpar();
283 }
284 else
285 {
286 MudaJogadorAtual();
287 }
288
289 IncrementaRodada();
290 }
